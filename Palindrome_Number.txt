class Solution {
public:
    bool isPalindrome(int x) {
        
        int end = 0;
        if(x%10 == 0 && x != 0)
            return false;
        while(end < x)
        {
            end = end*10 + x%10;
            x = x/10;
        }
        
        if(x == end || end/10 == x)
            return true;
        return false;
    }
};
